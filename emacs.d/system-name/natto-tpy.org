                    (remove-if-not
                     (lambda (f)
                       (with-temp-buffer
                         (insert-file-contents f)
                         (string-match "^type: text/org$"
                                       (first (split-string (buffer-string) "\n\n" t)))))
                     (mapcar
                      (lambda (f) (concat base-dir "/" f))
                      (directory-files base-dir
                                       nil "^[^\\$].+\\.tid$")))))))
